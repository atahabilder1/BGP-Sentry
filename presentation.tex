\begin{frame}
\frametitle{Introduction: The Problem}

\textbf{BGP lacks mandatory, global authentication.}

\vspace{0.3cm}

\begin{itemize}
    \item BGP route announcements are accepted without verification
    \item RPKI is the most widely accepted authentication mechanism
    \item However, only \textbf{37\%} of ASes have RPKI enabled
    \item \textbf{63\%} of Internet routing remains unverifiable
\end{itemize}

\vspace{0.3cm}

\textbf{Real-World BGP Hijacking Attacks:}
\begin{itemize}
    \item YouTube (2008): Pakistan Telecom hijacked YouTube's prefix~\cite{youtube2008hijack}
    \item MyEtherWallet (2018): \$150K+ stolen via BGP hijack~\cite{myetherwallet2018}
    \item Amazon Route 53 (2018): Cryptocurrency theft via BGP hijack~\cite{amazonroute53_2018}
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Motivation: Why Existing Solutions Fail}

\small
\begin{columns}[T]
\begin{column}{0.30\textwidth}
\textbf{RPKI Limitations}
\begin{itemize}\setlength{\itemsep}{2pt}
    \item Limited coverage
    \item Policy barriers
    \item No audit trail
    \item Cannot assess non-RPKI
\end{itemize}
\end{column}

\begin{column}{0.34\textwidth}
\textbf{Prior Blockchain\\(Onboarding)}
\begin{itemize}\setlength{\itemsep}{2pt}
    \item Trust-based onboarding
    \item No identity verification
    \item Dishonest nodes can join
    \item Sybil vulnerability
\end{itemize}
\end{column}

\begin{column}{0.34\textwidth}
\textbf{Prior Blockchain\\(Consensus)}
\begin{itemize}\setlength{\itemsep}{2pt}
    \item Asymmetric observations
    \item Assumes symmetric views
    \item No scalable consensus
    \item No persistent intelligence
\end{itemize}
\end{column}
\end{columns}

\vspace{0.5cm}

\centering
\fbox{\parbox{12cm}{\centering \textbf{Key Insight:} RPKI will never achieve 100\% coverage.\\We need to extend trust to the 63\% outside RPKI.}}

\end{frame}

% SLIDE 4: Three Challenges
% ==================================================
\begin{frame}
\frametitle{Three Critical Challenges}

\begin{enumerate}
    \item \textbf{Challenge 1: Incomplete RPKI Adoption}
    \begin{itemize}
        \item Only 37\% of ASes enforce RPKI
        \item Economic misalignment causes free-riding
        \item 63\% of routing remains unverifiable
    \end{itemize}

    \vspace{0.3cm}

    \item \textbf{Challenge 2: No Authenticated Observers}
    \begin{itemize}
        \item Blockchain validators are randomly selected
        \item May include malicious nodes
        \item Circular trust problem
    \end{itemize}

    \vspace{0.3cm}

    \item \textbf{Challenge 3: Limited Scalable Consensus}
    \begin{itemize}
        \item BGP observations are asymmetric across vantage points
        \item Traditional consensus assumes symmetric views
        \item No persistent security intelligence
    \end{itemize}
\end{enumerate}

\end{frame}

\begin{frame}
\frametitle{RPKI-Verified Onboarding}

\small
\textbf{How does an AS join the BGP-Sentry observer network?}

\vspace{0.2cm}

\centering
\begin{tabular}{c}
\fbox{\textbf{Regional Internet Registry (RIR)}} \\
{\scriptsize ARIN $\cdot$ RIPE $\cdot$ APNIC $\cdot$ LACNIC $\cdot$ AFRINIC} \\[0.2cm]
$\downarrow$ \quad {\scriptsize Issues RPKI certificate binding AS $\leftrightarrow$ IP resources} \\[0.2cm]
\fbox{\textbf{AS publishes Route Origin Authorizations (ROAs)}} \\
{\scriptsize prefix $\rightarrow$ authorized origin AS, stored in RPKI repository} \\[0.2cm]
$\downarrow$ \quad {\scriptsize rov-collector crawls all 5 RIR repositories} \\[0.2cm]
\fbox{\textbf{rov-collector Classification}} \\
{\scriptsize Measures real-world RPKI deployment per AS} \\[0.2cm]
$\downarrow$ \quad {\scriptsize Produces \texttt{as\_classification.json}} \\[0.2cm]
\fbox{\textbf{RPKINodeRegistry}} $\rightarrow$ \fbox{RPKI Validator} \quad \fbox{Non-RPKI (Monitored)} \\
{\scriptsize Onboarded as blockchain voter} \hspace{0.5cm} {\scriptsize Tracked via trust ratings} \\
\end{tabular}

\vspace{0.2cm}

\flushleft
\footnotesize
\textbf{Sybil resistance:} Creating a fake validator requires a real RPKI certificate from a RIR, which requires owning real IP address space --- economically infeasible.

\end{frame}

% SLIDE 10: Post-Hoc Forensic Analysis
% ==================================================
\begin{frame}
\frametitle{Post-Hoc Forensic Analysis}

\textbf{Why:} Real-time detection catches attacks as they happen.\\
Post-hoc analysis \textit{reconstructs the full history} from immutable blockchain records.

\vspace{0.3cm}

\small
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Who runs it}
\begin{itemize}\setlength{\itemsep}{2pt}
    \item Network operators (NOC teams)
    \item Security researchers / CERT teams
    \item Automated scheduled jobs (cron)
    \item Regulatory auditors
\end{itemize}

\vspace{0.2cm}

\textbf{How it runs}
\begin{itemize}\setlength{\itemsep}{2pt}
    \item Queries the blockchain offline
    \item No live network access needed
    \item Reproducible from any chain replica
\end{itemize}
\end{column}

\begin{column}{0.50\textwidth}
\textbf{Three analysis modules:}

\vspace{0.1cm}

\textbf{1. Blockchain Forensics}\\
{\footnotesize Attacker profiling, prefix history, observer cross-reference, audit trail generation}

\vspace{0.2cm}

\textbf{2. Targeted Attack Analyzer}\\
{\footnotesize Single-witness patterns, consensus escalation detection, temporal clustering}

\vspace{0.2cm}

\textbf{3. Longitudinal Behavior Analysis}\\
{\footnotesize Trust score trajectories, rating drift, repeat offenders, BGPCoin economy health}
\end{column}
\end{columns}

\end{frame}
